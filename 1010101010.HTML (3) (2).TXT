<script>
/* ===== CONTEXT MEMORY ===== */
const context = {
  lastQuery: "",
  interactions: 0,
  activeAgent: null
};

/* ===== AGENTS ===== */
const agents = [
  {emoji:"🧭", role:"Navigator", action:navigate},
  {emoji:"📡", role:"Scout", action:webSearch},
  {emoji:"🧠", role:"Synthesizer", action:summarize}
];

/* ===== LOG ===== */
const log = document.getElementById("log");

function write(title, content) {
  const e = document.createElement("div");
  e.className = "entry";
  e.innerHTML = `<strong>${title}</strong><br>${content}`;
  log.prepend(e);
}

/* ===== DELEGATION ===== */
function delegate() {
  const q = document.getElementById("query").value.trim();
  if (!q) return;

  context.lastQuery = q;
  context.interactions++;

  write("👁 YOU", q);

  // swarm decides
  agents.forEach((a,i)=>{
    setTimeout(()=>spawnAgent(a,q), i*600);
  });
}

/* ===== AGENT SPAWN ===== */
function spawnAgent(agent, query) {
  const el = document.createElement("div");
  el.className = "agent";
  el.textContent = agent.emoji;
  el.style.left = Math.random()*innerWidth+"px";
  el.style.top = (60 + Math.random()*(innerHeight-200))+"px";

  el.onclick = () => agent.action(query);

  document.body.appendChild(el);

  setTimeout(()=>el.remove(), 12000);
}

/* ===== AGENT ACTIONS ===== */
function navigate(q) {
  write("🧭 Navigator",
    "Determining intent… You are likely seeking understanding, resources, or direction.");
}

function summarize(q) {
  write("🧠 Synthesizer",
    "Once scouts return data, I will compress it into insight.");
}

/* ===== REAL WEB SEARCH (DuckDuckGo IA) ===== */
async function webSearch(q) {
  write("📡 Scout", "Searching the web for: <em>"+q+"</em>");

  const url = `https://api.duckduckgo.com/?q=${encodeURIComponent(q)}&format=json&no_redirect=1&no_html=1&skip_disambig=1`;

  try {
    const res = await fetch(url);
    const data = await res.json();

    let output = "";

    if (data.AbstractText) {
      output += `<p>${data.AbstractText}</p>`;
    }

    if (data.RelatedTopics) {
      data.RelatedTopics.slice(0,5).forEach(t=>{
        if(t.Text && t.FirstURL){
          output += `<p>🔗 <a href="${t.FirstURL}" target="_blank">${t.Text}</a></p>`;
        }
      });
    }

    if (!output) output = "No instant answers found, but deeper search may help.";

    write("📡 Scout Results", output);

  } catch(e) {
    write("📡 Scout Error", "Search failed. Network or API limitation.");
  }
}
